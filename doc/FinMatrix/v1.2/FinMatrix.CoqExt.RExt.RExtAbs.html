|a| <= b -> - b <= a <= b
Lemma Rabs_le_rev : forall a b : R, |a| <= b -> - b <= a <= b.
Proof.
  intros. bdestruct (a <? 0).
  - assert (|a| = -a); ra.
  - assert (|a| = a); ra.
    apply Rabs_pos_imply; ra.
Qed.
