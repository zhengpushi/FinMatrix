  a1 = b1 -> a2 = b2 -> a1 && a2 = b1 && b2.

Ltac solve_bool :=
  let H := fresh "H" in
  match goal with
  
  | |- ?b = false -> ?b <> true => intros H; rewrite H; auto
  
  | |- ?b <> true -> ?b = false => intros; apply not_true_is_false; auto
  end.
